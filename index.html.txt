<script>
    const scriptURL = "https://script.google.com/macros/s/AKfycbw4CJdIi24Hnywnn7txSnL6CZL2c9Zj9CVGm1HcaWE2skYMbjLNOk4uw2ORIjLJ28Qt/exec";
    const themes = {
        "Peacock": { primary: "#008080", hover: "#005959", bg: "#e0f2f1", lightBg: "#f0faf9" },
        "Blue":    { primary: "#0056b3", hover: "#003d82", bg: "#e6f2ff", lightBg: "#f0f7ff" },
        "Red":     { primary: "#dc3545", hover: "#a71d2a", bg: "#fcebec", lightBg: "#fdf3f4" },
        "Green":   { primary: "#28a745", hover: "#1e7e34", bg: "#e8f5e9", lightBg: "#f1f8f1" },
        "Orange":  { primary: "#fd7e14", hover: "#c6600d", bg: "#fff2e6", lightBg: "#fff8f0" },
        "Purple":  { primary: "#6f42c1", hover: "#59339d", bg: "#f3e8ff", lightBg: "#f9f4ff" },
        "Dark":    { primary: "#343a40", hover: "#23272b", bg: "#e9ecef", lightBg: "#f8f9fa" },
        "Teal":    { primary: "#20c997", hover: "#17a2b8", bg: "#e0fafa", lightBg: "#f0fcfc" },
        "Brown":   { primary: "#795548", hover: "#5d4037", bg: "#efebe9", lightBg: "#f5f3f1" },
        "Pink":    { primary: "#e83e8c", hover: "#c23373", bg: "#fcecf3", lightBg: "#fdf4f7" }
    };
    function applyTheme(t) { let s = themes[t] || themes["Peacock"]; document.documentElement.style.setProperty('--primary', s.primary); document.documentElement.style.setProperty('--hover', s.hover); document.documentElement.style.setProperty('--bg', s.bg); document.documentElement.style.setProperty('--light-bg', s.lightBg); }
    window.onload = function() {
        fetch(scriptURL + "?checkStatus=true").then(r => r.json()).then(data => {
            document.getElementById("loadingOverlay").classList.add("hidden");
            if(data.thumbnail) { 
                let img = document.getElementById("dynamicThumbnail");
                let fId = data.thumbnail.split('/d/')[1]?.split('/')[0];
                if(fId) { img.src = "https://lh3.googleusercontent.com/d/" + fId; img.style.display = "block"; }
            }
            document.getElementById("dynamicHeading").innerHTML = data.heading;
            document.getElementById("dynamicSubheading").innerHTML = data.subheading;
            applyTheme(data.theme);
            if(data.status === "OFF") document.getElementById("linkOffMessage").classList.remove("hidden");
            else document.getElementById("dataForm").classList.remove("hidden");
        });
    };
    function showChainFields() { const c = document.getElementById("chainType").value; document.getElementById("all-fields").classList.remove("hidden"); document.getElementById("central-fields").classList.toggle("hidden", c !== "central"); document.getElementById("department-fields").classList.toggle("hidden", c !== "department"); }
    
    function fetchDistrict() { 
        let pInput = document.getElementById("pincode");
        fetch(scriptURL + "?pincode=" + pInput.value).then(r => r.json()).then(d => { 
            document.getElementById("district").value = d.district; 
            document.getElementById("division").value = d.division; 
            document.getElementById("state").value = d.state; 
            document.getElementById("region").value = d.region; 
            
            if (d.district === "Not Found" || d.district === "") {
                pInput.setCustomValidity("Yeh Pincode record mein nahi hai. Kripya sahi Pincode add karein.");
                pInput.reportValidity();
            } else {
                pInput.setCustomValidity("");
            }
        }); 
    }
    
    function submitForm(e){
e.preventDefault();
const btn=document.getElementById("submitBtn");
if(btn.disabled) return false;
btn.disabled=true;
btn.innerText="Submitting...";
document.getElementById("submitStatus").innerText="Please wait...";
fetch(scriptURL,{method:"POST",body:JSON.stringify({ chainType: document.getElementById("chainType").value, userName: document.getElementById("userName").value, contactNumber: document.getElementById("contactNumber").value, nigranLevel: document.getElementById("nigranLevel").value, zimmedarLevel: document.getElementById("zimmedarLevel").value, departmentList: document.getElementById("departmentList").value, risalaReport: document.getElementById("risalaReport").value, pincode: document.getElementById("pincode").value, district: document.getElementById("district").value, division: document.getElementById("division").value, state: document.getElementById("state").value, region: document.getElementById("region").value })}).then(r=>r.json()).then(d=>{
 if(d.status==="Success"){
  document.getElementById("dataForm").classList.add("hidden");
  document.getElementById("successMessage").classList.remove("hidden");
 }else{
  alert(d.message||d.status);
  btn.disabled=false;
  btn.innerText="Submit Report";
  document.getElementById("submitStatus").innerText="";
 }
}).catch(()=>{
 alert("Network error. Please try again.");
 btn.disabled=false;
 btn.innerText="Submit Report";
 document.getElementById("submitStatus").innerText="";
});
}
